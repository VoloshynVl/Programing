#include <stdio.h>

int main() {
    int n;

    // Введення натурального числа n
    printf("Ведіть число (4 < n < 10: ");
    scanf("%d", &n);

    // Перевірка умови для n
    if (n <= 4 || n >= 10) {
        printf("Неправильне значення n. Будь ласка, введіть число в межах (4 < n < 10).\n");
        return 1; // Помилка введення
    }

    // Введення дійсних чисел a1, a2, ..., an
    double a[n];
    printf("Введіть %d дійсних чисел:\n", n);
    for (int i = 0; i < n; i++) {
        printf("a%d: ", i + 1);
        scanf("%lf", &a[i]);
    }

    // Обчислення виразу
    double result = 0;
    for (int i = 0; i < n; i++) {
        result += ((i % 2 == 0) ? 1 : -1) * a[i];
    }

    // Виведення результату
    result *= 2;
    printf("Результат обчислення виразу: %.2lf\n", result);

    return 0;
}
