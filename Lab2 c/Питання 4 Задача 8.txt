#include <stdio.h>

int main() {
    double V, U, T1, T2, S1, S2, S;

    // Введення даних користувачем
    printf("Введіть швидкість човна в стоячій воді (км/год): ");
    scanf("%lf", &V);
    printf("Введіть швидкість течії річки (км/год): ");
    scanf("%lf", &U);
    printf("Введіть час руху човна по озеру (години): ");
    scanf("%lf", &T1);
    printf("Введіть час руху човна по річці проти течії (години): ");
    scanf("%lf", &T2);

    // Обчислення шляху в стоячій воді та по річці
    S1 = V * T1;
    S2 = (V - U) * T2;

    // Загальний шлях пройдений човном
    S = S1 + S2;

    // Виведення результату
    printf("Шлях пройдений човном: %.2lf км\n", S);

    return 0;
}
