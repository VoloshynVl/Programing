#include <stdio.h>

int main() {
    int a, b, c; // Змінні
    int *sh_a, *sh_b, *sh_c; // Покажчики

    // Введення значень для a, b і c
    printf("Введіть значення для a: ");
    scanf("%d", &a);
    printf("Введіть значення для b: ");
    scanf("%d", &b);
    printf("Введіть значення для c: ");
    scanf("%d", &c);

    // Направлення покажчиків на a, b і c
    sh_a = &a;
    sh_b = &b;
    sh_c = &c;

    // Збільшення значення a в 2 рази через покажчик
    *sh_a = *sh_a * 2;

    // Поміняти місцями значення b і c через покажчики
    int tp = *sh_b;
    *sh_b = *sh_c;
    *sh_c = tp;

    // Виведення результату
    printf("Нове значення a: %d\n", a);
    printf("Нове значення b: %d\n", b);
    printf("Нове значення c: %d\n", c);

    return 0;
}
